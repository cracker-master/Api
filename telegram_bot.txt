def db_init():
    con = sqlite3.connect(DB_PATH)
    cur = con.cursor()
    cur.executescript("""
        CREATE TABLE IF NOT EXISTS users (
            user_id     INTEGER PRIMARY KEY,
            username    TEXT,
            first_name  TEXT,
            joined_at   TEXT,
            cmd_count   INTEGER DEFAULT 0,
            last_seen   TEXT
        );
        CREATE TABLE IF NOT EXISTS cmd_log (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id     INTEGER,
            command     TEXT,
            args        TEXT,
            ts          TEXT
        );
    """)
    con.commit()
    con.close()

def db_upsert_user(user):
    con = sqlite3.connect(DB_PATH)
    con.execute("""
        INSERT INTO users(user_id, username, first_name, joined_at, cmd_count, last_seen)
        VALUES(?,?,?,?,0,?)
        ON CONFLICT(user_id) DO UPDATE SET
            username=excluded.username,
            last_seen=excluded.last_seen,
            cmd_count=cmd_count+1
    """, (user.id, user.username, user.first_name,
          datetime.now().isoformat(), datetime.now().isoformat()))
    con.commit()
    con.close()

def db_log_cmd(user_id, command, args=""):
    con = sqlite3.connect(DB_PATH)
    con.execute("INSERT INTO cmd_log(user_id,command,args,ts) VALUES(?,?,?,?)",
                (user_id, command, args, datetime.now().isoformat()))
    con.commit()
    con.close()

def db_get_stats():
    con = sqlite3.connect(DB_PATH)
    users = con.execute("SELECT COUNT(*) FROM users").fetchone()[0]
    cmds  = con.execute("SELECT COUNT(*) FROM cmd_log").fetchone()[0]
    top   = con.execute(
        "SELECT command, COUNT(*) as c FROM cmd_log GROUP BY command ORDER BY c DESC LIMIT 5"
    ).fetchall()
    all_users = con.execute("SELECT user_id, username, first_name, cmd_count, last_seen FROM users").fetchall()
    con.close()
    return users, cmds, top, all_users

# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
def is_rate_limited(user_id: int) -> bool:
    if user_id in ADMIN_IDS:
        return False
    now = time.time()
    history = rate_limits.get(user_id, [])
    history = [t for t in history if now - t < 60]
    rate_limits[user_id] = history
    if len(history) >= 10:
        return True
    rate_limits[user_id].append(now)
    return False

async def check_rate(update: Update) -> bool:

            parse_mode="Markdown"
        )
        return False
    return True

def stealth_get(url: str, timeout=15) -> requests.Response:
    """Anti-bot bypass: random UA + optional proxy + retry."""
    proxies = {}
    if PROXY_LIST:
        proxy = random.choice(PROXY_LIST)
        proxies = {"http": proxy, "https": proxy}

    for attempt in range(3):
        try:
            resp = requests.get(
                url,
                headers=get_stealth_headers(),
                proxies=proxies if proxies else None,
                timeout=timeout,
                allow_redirects=True
            )
            if resp.status_code == 403 and attempt < 2:
                time.sleep(1)
                continue
            return resp
        except Exception:
            if attempt == 2:
                raise
            time.sleep(1)

def groq_chat(messages: list, system: str = None, model="llama3-8b-8192") -> str:
    payload = []
    if system:
        payload.append({"role": "system", "content": system})
    payload.extend(messages)
    resp = requests.post(
        "https://api.groq.com/openai/v1/chat/completions",
        headers={"Authorization": f"Bearer {GROQ_KEY}", "Content-Type": "application/json"},
        json={"model": model, "messages": payload, "max_tokens": 1500, "temperature": 0.7},
        timeout=30
    )
    resp.raise_for_status()
    return resp.json()["choices"][0]["message"]["content"]

def groq_transcribe(audio_bytes: bytes, filename="voice.ogg") -> str:
    """Groq Whisper — FREE voice transcription."""
    resp = requests.post(
        "https://api.groq.com/openai/v1/audio/transcriptions",
        headers={"Authorization": f"Bearer {GROQ_KEY}"},
        files={"file": (filename, audio_bytes, "audio/ogg")},
        data={"model": "whisper-large-v3", "response_format": "text"},
        timeout=60
    )
    resp.raise_for_status()
    return resp.text.strip()

async def apidoc(update: Update, context: ContextTypes.DEFAULT_TYPE):

    db_upsert_user(update.effective_user)
    if not context.args:
        await update.message.reply_text(
            "⚠️ API URL দাও।\n`/apidoc https://jsonplaceholder.typicode.com/posts/1`",
            parse_mode="Markdown"
        )
        return

    url = context.args[0]
    if not url.startswith("http"): url = "https://" + url
    db_log_cmd(update.effective_user.id, "/apidoc", url)

        parse_mode="Markdown"
    )

    try:
        # GET request
        resp = stealth_get(url)
        resp.raise_for_status()

        # Method detect
        methods_available = ["GET"]
        try:
            head_resp = requests.head(url, headers=get_stealth_headers(), timeout=5)
            allow = head_resp.headers.get("Allow", "")
            if allow:
                methods_available = [m.strip() for m in allow.split(",")]
        except Exception:
            pass

        # Response parse
        try:
            json_data = resp.json()
            data_str = json.dumps(json_data, indent=2, ensure_ascii=False)[:2000]
            dtype = "JSON"
        except Exception:
            data_str = resp.text[:2000]
            dtype = "Text"

            f"নিচের API এর জন্য বাংলায় একটি সম্পূর্ণ documentation তৈরি করো:\n\n"
            f"**URL:** {url}\n"
            f"**Method:** {', '.join(methods_available)}\n"
            f"**Response Type:** {dtype}\n"
            f"**Status Code:** {resp.status_code}\n"
            f"**Response Headers:** {dict(list(resp.headers.items())[:5])}\n"
            f"**Response Body (sample):**\n```\n{data_str}\n```\n\n"
            "Documentation এ অন্তর্ভুক্ত করো:\n"
            "1. 📌 API Overview (কী কাজ করে)\n"
            "2. 🔗 Endpoint details\n"
            "3. 📥 Request format (headers, params)\n"
            "4. 📤 Response fields (প্রতিটি field এর বর্ণনা)\n"
            "5. 💻 Code example (Python ও curl)\n"
            "6. ⚠️ সম্ভাব্য errors\n"
            "সহজ ও বিস্তারিত ভাষায় লেখো।"
        )

        doc = groq_chat(
            messages=[{"role": "user", "content": doc_prompt}],
            system="তুমি একজন API documentation বিশেষজ্ঞ। সম্পূর্ণ ও বিস্তারিত documentation তৈরি করো।",
            model="llama3-8b-8192"
        )

        if len(doc) > 3800:
            await thinking.edit_text(

    except Exception as e:

# /watch — Scheduled Auto Scrape & Alert
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
async def _watch_task(url: str, interval: int, chat_id: int, watch_id: str, bot):
    last_content = None
    count = 0
    try:
        await bot.send_message(
            chat_id=chat_id,
            text=(
                f"👀 *Watch শুরু হয়েছে!*\n"
                f"🆔 ID: `{watch_id}`\n"
                f"🔗 `{url}`\n"
                f"⏱ প্রতি `{interval}` সেকেন্ডে check হবে\n"
                f"_বন্ধ করতে:_ `/watchstop {watch_id}`"
            ),
            parse_mode="Markdown"
        )

        while True:
            await asyncio.sleep(interval)
            count += 1
            try:
                resp = stealth_get(url, timeout=10)
                current = resp.text[:500]

                if last_content is None:
                    last_content = current
                    await bot.send_message(
                        chat_id=chat_id,
                        text=f"📸 *{watch_id}* — প্রথম snapshot নেওয়া হয়েছে। পরিবর্তন হলে জানাবো।",
                        parse_mode="Markdown"
                    )
                elif current != last_content:
                    await bot.send_message(
                        chat_id=chat_id,
                        text=(
                            f"🔔 *পরিবর্তন ধরা পড়েছে! [{watch_id}]*\n"
                            f"🔗 `{url}`\n"
                            f"⏰ {datetime.now().strftime('%H:%M:%S')}\n"
                            f"🔄 Check #{count}\n\n"
                            f"*নতুন content (প্রথম ৩০০ অক্ষর):*\n`{current[:300]}`"
                        ),
                        parse_mode="Markdown"
                    )
                    last_content = current

            except Exception as e:
                await bot.send_message(
                    chat_id=chat_id,
                    text=f"⚠️ *{watch_id}* Check #{count} Error: `{str(e)}`",
                    parse_mode="Markdown"
                )

    except asyncio.CancelledError:
        await bot.send_message(
            chat_id=chat_id,
            text=f"🛑 *{watch_id}* বন্ধ। মোট {count}টি check হয়েছিল।",
            parse_mode="Markdown"
        )
    finally:
        watch_tasks.pop(watch_id, None)

        )
        return
    url = context.args[0]

    except ValueError:

        unique = [c for c in api_calls if not (c["url"] in seen or seen.add(c["url"]))]

async def network_monitor(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not await check_rate(update): return
    db_upsert_user(update.effective_user)
    if not context.args:
        await update.message.reply_text("⚠️ URL দাও। যেমন: `/network https://example.com`", parse_mode="Markdown")
        return
    url = context.args[0]
    if not url.startswith("http"): url = "https://" + url
    db_log_cmd(update.effective_user.id, "/network", url)
    thinking = await update.message.reply_text(f"🔎 *Network Monitor...*\n`{url}`", parse_mode="Markdown")
    buckets = {"XHR": [], "FETCH": [], "WEBSOCKET": [], "SCRIPT": [], "OTHER": []}
    try:
        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            page = await browser.new_page()
            page.on("websocket", lambda ws: buckets["WEBSOCKET"].append(f"`WS` `{ws.url[:100]}`"))
            def on_req(req):
                rt = req.resource_type.upper()
                entry = f"`{req.method}` `{req.url[:100]}`"
                if rt in buckets: buckets[rt].append(entry)
                elif rt not in ["IMAGE", "MEDIA", "FONT", "STYLESHEET"]: buckets["OTHER"].append(entry)
            page.on("request", on_req)
            await playwright_stealth_goto(page, url)
            await page.wait_for_timeout(2000)
            await browser.close()
        result = f"🔎 *Network — {urlparse(url).netloc}*\n\n"
        for rtype, entries in buckets.items():
            if entries:
                result += f"*{rtype} ({len(entries)}):*\n"
                for e in entries[:4]: result += f"  {e}\n"
                if len(entries) > 4: result += f"  _+{len(entries)-4}টি_\n"
                result += "\n"
        total = sum(len(v) for v in buckets.values())
        result += f"📊 *মোট: {total}*"

# WebSocket Monitor
# ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
async def _ws_monitor_task(ws_url, chat_id, monitor_id, bot, max_messages=50):
    received = 0
    try:
        await bot.send_message(chat_id=chat_id,
            text=f"📡 *WebSocket Connected!*\n🆔 `{monitor_id}`\n`{ws_url}`\n_বন্ধ: `/wsstop {monitor_id}`_",
            parse_mode="Markdown")
        async with websockets.connect(ws_url, ping_interval=20, extra_headers={"User-Agent": get_stealth_headers()["User-Agent"]}) as ws:
            async for raw_msg in ws:
                received += 1
                try:
                    data = json.loads(raw_msg)
                    fmt = json.dumps(data, indent=2, ensure_ascii=False)
                    if len(fmt) > 800: fmt = fmt[:800] + "\n...(cut)"
                    text = f"📨 *#{received} [{monitor_id}]*\n```json\n{fmt}\n```"
                except Exception:
                    text = f"📨 *#{received} [{monitor_id}]*\n`{str(raw_msg)[:500]}`"
                await bot.send_message(chat_id=chat_id, text=text, parse_mode="Markdown")
                if received >= max_messages:
                    await bot.send_message(chat_id=chat_id, text=f"⚠️ {monitor_id} — {max_messages}টি message পাওয়া গেছে।", parse_mode="Markdown")
                    break
    except asyncio.CancelledError:
        await bot.send_message(chat_id=chat_id, text=f"🛑 *{monitor_id}* বন্ধ। ({received}টি)", parse_mode="Markdown")
    except Exception as e:
        await bot.send_message(chat_id=chat_id, text=f"❌ *{monitor_id}* Error: `{str(e)}`", parse_mode="Markdown")
    finally:
        ws_monitors.pop(monitor_id, None)

async def ws_monitor(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not await check_rate(update): return
    db_upsert_user(update.effective_user)
    if not context.args:
        await update.message.reply_text("⚠️ URL দাও।\n`/ws wss://stream.binance.com:9443/ws/btcusdt@trade`", parse_mode="Markdown")
        return
    url = context.args[0]
    if url.startswith("wss://") or url.startswith("ws://"):
        ws_counter["n"] += 1
        mid = f"WS{ws_counter['n']}"
        task = asyncio.create_task(_ws_monitor_task(url, update.effective_chat.id, mid, context.bot))
        ws_monitors[mid] = task
        await update.message.reply_text(f"🚀 *Monitor শুরু!* 🆔 `{mid}`\n`{url}`", parse_mode="Markdown")
        return
    if not url.startswith("http"): url = "https://" + url
    thinking = await update.message.reply_text(f"🔍 *WebSocket খোঁজা হচ্ছে...*\n`{url}`", parse_mode="Markdown")
    found = []
    try:
        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            page = await browser.new_page()
            page.on("websocket", lambda ws: found.append(ws.url) if ws.url not in found else None)
            page.on("request", lambda r: found.append(r.url) if r.url.startswith("ws") and r.url not in found else None)
            await playwright_stealth_goto(page, url)
            await page.wait_for_timeout(5000)
            await browser.close()
        if not found:
            await thinking.edit_text("⚠️ WebSocket পাওয়া যায়নি।\nDirect WS URL দিলে: `/ws wss://...`", parse_mode="Markdown")
            return
        result = f"✅ *{len(found)}টি WebSocket পাওয়া গেছে!*\n\n"
        for i, u in enumerate(found, 1): result += f"`{i}.` `{u}`\n"
        result += "\nConnect করতে: `/ws <url>`"
        await thinking.edit_text(result, parse_mode="Markdown")
    except Exception as e:
        await thinking.edit_text(f"❌ Error: `{str(e)}`", parse_mode="Markdown")

async def scrape(update: Update, context: ContextTypes.DEFAULT_TYPE):

        return
    url = context.args[0]
    if not url.startswith("http"): url = "https://" + url
    db_log_cmd(update.effective_user.id, "/scrape", url)
    try:
        resp = stealth_get(url)
        soup = BeautifulSoup(resp.text, "html.parser")
        title = soup.title.string.strip() if soup.title else "N/A"
        desc = (soup.find("meta", attrs={"name": "description"}) or {}).get("content", "N/A")
        links = [a["href"] for a in soup.find_all("a", href=True)]
        ext = [l for l in links if l.startswith("http") and urlparse(url).netloc not in l]
        intr = [l for l in links if not l.startswith("http")]
        imgs = [i.get("src","") for i in soup.find_all("img") if i.get("src")]
        result = (f"🌐 *{urlparse(url).netloc}*\n📌 *Title:* {title[:80]}\n📝 *Desc:* {str(desc)[:120]}\n"
                  f"📊 Status: `{resp.status_code}`\n🔗 Int:{len(intr)} Ext:{len(ext)} 🖼{len(imgs)}\n\n*External:*\n")
        for l in ext[:5]: result += f"`{l[:100]}`\n"
        await update.message.reply_text(result, parse_mode="Markdown")
    except Exception as e:
        await update.message.reply_text(f"❌ Error: `{str(e)}`", parse_mode="Markdown")

# User personas DB এ save করা হবে
def db_init_personas():
    con = sqlite3.connect(DB_PATH)
    con.execute("""
        CREATE TABLE IF NOT EXISTS user_personas (
            user_id     INTEGER PRIMARY KEY,
            persona_id  TEXT DEFAULT 'default',
            custom_name TEXT,
            custom_emoji TEXT,
            custom_prompt TEXT,
            created_at  TEXT
        )
    """)
    con.commit()
    con.close()

def db_get_persona(user_id: int) -> dict:
    con = sqlite3.connect(DB_PATH)
    row = con.execute(
        "SELECT persona_id, custom_name, custom_emoji, custom_prompt FROM user_personas WHERE user_id=?",
        (user_id,)
    ).fetchone()
    con.close()
    if not row:
        return BUILTIN_PERSONAS["default"]
    pid, cname, cemoji, cprompt = row
    if pid == "custom" and cprompt:
        return {"name": cname or "Custom AI", "emoji": cemoji or "🤖", "prompt": cprompt}
    return BUILTIN_PERSONAS.get(pid, BUILTIN_PERSONAS["default"])

def db_set_persona(user_id: int, persona_id: str, custom_name=None, custom_emoji=None, custom_prompt=None):
    con = sqlite3.connect(DB_PATH)
    con.execute("""
        INSERT INTO user_personas(user_id, persona_id, custom_name, custom_emoji, custom_prompt, created_at)
        VALUES(?,?,?,?,?,?)
        ON CONFLICT(user_id) DO UPDATE SET
            persona_id=excluded.persona_id,
            custom_name=excluded.custom_name,
            custom_emoji=excluded.custom_emoji,
            custom_prompt=excluded.custom_prompt,
            created_at=excluded.created_at
    """, (user_id, persona_id, custom_name, custom_emoji, custom_prompt, datetime.now().isoformat()))
    con.commit()
    con.close()

import os, shutil, zipfile, tempfile, subprocess

def parse_github_url(url: str):
    """GitHub URL থেকে owner ও repo বের করো।"""
    url = url.rstrip("/").replace(".git", "")
    # https://github.com/owner/repo
    match = re.match(r"(?:https?://)?github\.com/([^/]+)/([^/]+)", url)
    if match:
        return match.group(1), match.group(2)
    # owner/repo shorthand
    parts = url.split("/")
    if len(parts) == 2:
        return parts[0], parts[1]
    return None, None

async def github_repo_info(owner: str, repo: str) -> dict:
    """GitHub API দিয়ে repo info আনো (ফ্রি, no key needed)."""
    resp = requests.get(
        f"https://api.github.com/repos/{owner}/{repo}",
        headers={"Accept": "application/vnd.github.v3+json"},
        timeout=10
    )
    if resp.status_code == 404:
        raise ValueError("Repo পাওয়া যায়নি। URL সঠিক আছে?")
    resp.raise_for_status()
    return resp.json()

async def clone_and_zip(owner: str, repo: str, branch: str = None) -> tuple[str, str]:
    """
    GitHub repo zip download করো, temp dir এ রাখো।
    Returns: (zip_path, temp_dir)
    """
    # Branch না দিলে default branch নাও
    if not branch:
        info = requests.get(
            f"https://api.github.com/repos/{owner}/{repo}",
            headers={"Accept": "application/vnd.github.v3+json"},
            timeout=10
        ).json()
        branch = info.get("default_branch", "main")

    zip_url = f"https://github.com/{owner}/{repo}/archive/refs/heads/{branch}.zip"
    resp = requests.get(zip_url, stream=True, timeout=60)

    if resp.status_code == 404:
        # main এ না পেলে master try করো
        zip_url = f"https://github.com/{owner}/{repo}/archive/refs/heads/master.zip"
        resp = requests.get(zip_url, stream=True, timeout=60)
        resp.raise_for_status()

    tmp_dir = tempfile.mkdtemp()
    zip_path = os.path.join(tmp_dir, f"{repo}.zip")

    with open(zip_path, "wb") as f:
        for chunk in resp.iter_content(chunk_size=8192):
            f.write(chunk)

    return zip_path, tmp_dir


# /clone — GitHub Repo Clone & Download
async def github_clone(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not await check_rate(update): return
    db_upsert_user(update.effective_user)

    if not context.args:
        await update.message.reply_text(
            "⚠️ GitHub URL দাও।\n\n"
            "*উদাহরণ:*\n"
            "`/clone https://github.com/owner/repo`\n"
            "`/clone owner/repo`\n"
            "`/clone owner/repo main`  ← branch নির্দিষ্ট করতে\n\n"
            "📦 বট repo zip করে পাঠাবে!",
            parse_mode="Markdown"
        )
        return

    raw_url = context.args[0]
    branch  = context.args[1] if len(context.args) > 1 else None
    owner, repo = parse_github_url(raw_url)

    if not owner or not repo:
        await update.message.reply_text(
            "❌ URL সঠিক না।\n`/clone https://github.com/owner/repo`",
            parse_mode="Markdown"
        )
        return

    db_log_cmd(update.effective_user.id, "/clone", f"{owner}/{repo}")
    thinking = await update.message.reply_text(
        f"📦 *GitHub Repo Clone হচ্ছে...*\n"
        f"👤 `{owner}` / 📁 `{repo}`\n"
        f"⏳ অপেক্ষা করো...",
        parse_mode="Markdown"
    )

    tmp_dir = None
    try:
        # Repo info আনো
        info = await github_repo_info(owner, repo)
        size_kb = info.get("size", 0)
        stars   = info.get("stargazers_count", 0)
        forks   = info.get("forks_count", 0)
        lang    = info.get("language", "N/A")
        desc    = info.get("description", "N/A") or "N/A"
        def_br  = info.get("default_branch", "main")
        use_branch = branch or def_br

        # Size check — 50MB এর বেশি হলে না নামানো ভালো
        if size_kb > 50000:
            await thinking.edit_text(
                f"⚠️ *Repo অনেক বড়!* (`{size_kb//1024}MB`)\n\n"
                f"📊 *Repo Info:*\n"
                f"👤 Owner: `{owner}`\n"
                f"📁 Repo: `{repo}`\n"
                f"⭐ Stars: `{stars}` | 🍴 Forks: `{forks}`\n"
                f"💻 Language: `{lang}`\n"
                f"📝 {desc[:100]}\n\n"
                f"🔗 Download link:\n`https://github.com/{owner}/{repo}/archive/refs/heads/{use_branch}.zip`",
                parse_mode="Markdown"
            )
            return

        await thinking.edit_text(
            f"📥 *Download হচ্ছে...*\n"
            f"`{owner}/{repo}` [{use_branch}]\n"
            f"📦 Size: ~`{size_kb}KB`",
            parse_mode="Markdown"
        )

        # Download & zip
        zip_path, tmp_dir = await clone_and_zip(owner, repo, use_branch)
        zip_size = os.path.getsize(zip_path)

        # Telegram file limit: 50MB
        if zip_size > 50 * 1024 * 1024:
            await thinking.edit_text(
                f"⚠️ Zip ফাইল `{zip_size//1024//1024}MB` — Telegram limit (50MB) ছাড়িয়েছে।\n\n"
                f"🔗 সরাসরি download করো:\n"
                f"`https://github.com/{owner}/{repo}/archive/refs/heads/{use_branch}.zip`",
                parse_mode="Markdown"
            )
            return

        await thinking.delete()

        # Telegram এ পাঠাও
        caption = (
            f"📦 *{owner}/{repo}*\n"
            f"🌿 Branch: `{use_branch}`\n"
            f"⭐ {stars} | 🍴 {forks} | 💻 {lang}\n"
            f"📝 {desc[:80]}\n"
            f"📊 Size: `{zip_size//1024}KB`"
        )
        with open(zip_path, "rb") as zf:
            await update.message.reply_document(
                document=zf,
                filename=f"{repo}-{use_branch}.zip",
                caption=caption,
                parse_mode="Markdown"
            )

    except ValueError as e:
        await thinking.edit_text(f"❌ {str(e)}", parse_mode="Markdown")
    except Exception as e:
        await thinking.edit_text(f"❌ Error: `{str(e)}`", parse_mode="Markdown")
    finally:
        if tmp_dir and os.path.exists(tmp_dir):
            shutil.rmtree(tmp_dir, ignore_errors=True)

        country= loc.get("country", "")

        weather_resp = requests.get(
            "https://api.open-meteo.com/v1/forecast",
            params={
                "latitude": lat, "longitude": lon,
                "current": "temperature_2m,relative_humidity_2m,wind_speed_10m,weather_code,apparent_temperature,precipitation",
                "daily": "temperature_2m_max,temperature_2m_min,precipitation_sum,weather_code",

                "timezone": "auto"

        ).json()

        feels     = cur.get("apparent_temperature", "?")
        humidity  = cur.get("relative_humidity_2m", "?")
        wind      = cur.get("wind_speed_10m", "?")
        precip    = cur.get("precipitation", 0)
        wcode     = cur.get("weather_code", 0)

            if code == 0: return "☀️ পরিষ্কার"
            elif code <= 3: return "⛅ আংশিক মেঘলা"
            elif code <= 49: return "🌫️ কুয়াশা"
            elif code <= 67: return "🌧️ বৃষ্টি"
            elif code <= 77: return "❄️ তুষার"
            elif code <= 82: return "🌦️ বৃষ্টি"
            elif code <= 99: return "⛈️ ঝড়"
            return "🌤️"

        result = (
            f"🌍 *{name}, {country}*\n"
            f"━━━━━━━━━━━━━━━━━━━━\n"
            f"{weather_emoji(wcode)}\n\n"

            f"💧 আর্দ্রতা: `{humidity}%`\n"
            f"💨 বাতাস: `{wind} km/h`\n"
            f"🌧️ বৃষ্টি: `{precip} mm`\n\n"
            f"━━━━━━━━━━━━━━━━━━━━\n"
            f"📅 *আগামী ৩ দিন:*\n"
        )

        days = ["আজ","কাল","পরশু"]

            mn = daily["temperature_2m_min"][i]
            pr = daily["precipitation_sum"][i]
            wc = daily["weather_code"][i]
            result += f"{days[i]}: {weather_emoji(wc)} {mn}°~{mx}°C 🌧{pr}mm\n"

            f"  2nd: `{times[1]:.0f}ms`\n"
            f"  3rd: `{times[2]:.0f}ms`\n\n"
            f"📊 Avg: `{avg:.0f}ms` | Min: `{mn:.0f}ms` | Max: `{mx:.0f}ms`\n"
            f"{'🟢 Fast' if avg<200 else '🟡 Medium' if avg<500 else '🔴 Slow'}"
